class QueryPlanner:
    """
    Analyzes a ComplexQuery and decomposes it into an executable plan (PlannedQuery).
    """

    def __init__(self, db_manager: 'GrizabellaDBManager'):
        """
        Initializes the QueryPlanner.

        Args:
            db_manager: An instance of GrizabellaDBManager to access schema information.
        """
        self._db_manager = db_manager
        self._schema_manager = db_manager._schema_manager # Access via db_manager

    def plan(self, query: ComplexQuery) -> PlannedQuery:
        """
        Generates an execution plan for the given complex query.

        Args:
            query: The ComplexQuery object.

        Returns:
            A PlannedQuery object.
        Raises:
            ValueError: If the query is invalid or schema validation fails.
        """
        logger.info("Planning complex query: %s", query.description or "Untitled Query")

        if not query.components:
            raise ValueError("ComplexQuery must have at least one component.")

        # Determine the final target object type from the first component.
        final_target_object_type = query.components[0].object_type_name
        otd_final_target = self._schema_manager.get_object_type_definition(final_target_object_type)
        if not otd_final_target:
            raise ValueError(
                f"Primary target ObjectTypeDefinition '{final_target_object_type}' "
                "from first query component not found in schema."
            )

        all_component_plans: List[PlannedComponentExecution] = []

        for idx, component in enumerate(query.components):
            component_steps: List[PlannedStep] = []
            current_input_source_idx_for_component: Optional[int] = None
            component_object_type_name = component.object_type_name

            otd_component = self._schema_manager.get_object_type_definition(component_object_type_name)
            if not otd_component:
                raise ValueError(
                    f"QueryComponent {idx}: ObjectTypeDefinition '{component_object_type_name}' not found."
                )

            # 1. Relational Filters (SQLite)
            if component.relational_filters:
                for rel_filter in component.relational_filters:
                    prop_def = self._schema_manager.get_property_definition_for_object_type(
                        component_object_type_name, rel_filter.property_name
                    )
                    if not prop_def:
                        raise ValueError(
                            f"QueryComponent {idx}: Property '{rel_filter.property_name}' "
                            f"not found for ObjectType '{component_object_type_name}'."
                        )
                    # TODO: Add operator/value type compatibility check with prop_def.data_type

                component_steps.append(PlannedStep(
                    step_type="sqlite_filter",
                    details={
                        "object_type_name": component_object_type_name,
                        "filters": component.relational_filters
                    },
                    input_object_ids_source_step_index=None
                ))
                current_input_source_idx_for_component = len(component_steps) - 1

            # 2. Embedding Searches (LanceDB)
            if component.embedding_searches:
                for emb_search in component.embedding_searches:
                    emb_def = self._schema_manager.get_embedding_definition(emb_search.embedding_definition_name)
                    if not emb_def:
                        raise ValueError(
                            f"QueryComponent {idx}: EmbeddingDefinition "
                            f"'{emb_search.embedding_definition_name}' not found."
                        )
                    if emb_def.object_type_name != component_object_type_name:
                        raise ValueError(
                            f"QueryComponent {idx}: EmbeddingDefinition '{emb_search.embedding_definition_name}' "
                            f"is for ObjectType '{emb_def.object_type_name}', but component targets "
                            f"'{component_object_type_name}'."
                        )
                    # TODO: Validate query_vector dimension matches emb_def.dimensions

                    component_steps.append(PlannedStep(
                        step_type="lancedb_search",
                        details={
                            "embedding_search_clause": emb_search
                        },
                        input_object_ids_source_step_index=current_input_source_idx_for_component
                    ))
                    current_input_source_idx_for_component = len(component_steps) - 1
            
            # 3. Graph Traversals (Kuzu)
            if component.graph_traversals:
                for traversal in component.graph_traversals:
                    rtd = self._schema_manager.get_relation_type_definition(traversal.relation_type_name)
                    if not rtd:
                        raise ValueError(
                            f"QueryComponent {idx}: RelationTypeDefinition "
                            f"'{traversal.relation_type_name}' not found."
                        )
                    
                    if component_object_type_name not in rtd.source_object_type_names:
                        raise ValueError(
                            f"QueryComponent {idx}: Relation '{traversal.relation_type_name}' "
                            f"cannot originate from ObjectType '{component_object_type_name}'. "
                            f"Allowed sources: {rtd.source_object_type_names}."
                        )
                    if traversal.target_object_type_name not in rtd.target_object_type_names:
                        raise ValueError(
                            f"QueryComponent {idx}: Relation '{traversal.relation_type_name}' "
                            f"cannot target ObjectType '{traversal.target_object_type_name}'. "
                            f"Allowed targets: {rtd.target_object_type_names}."
                        )

                    if traversal.target_object_properties:
                        target_otd = self._schema_manager.get_object_type_definition(
                            traversal.target_object_type_name
                        )
                        if not target_otd:
                            raise ValueError(
                                f"QueryComponent {idx}: Target ObjectTypeDefinition "
                                f"'{traversal.target_object_type_name}' for graph traversal not found."
                            )
                        for rel_filter in traversal.target_object_properties:
                            prop_def = self._schema_manager.get_property_definition_for_object_type(
                                traversal.target_object_type_name, rel_filter.property_name
                            )
                            if not prop_def:
                                raise ValueError(
                                    f"QueryComponent {idx}: Property '{rel_filter.property_name}' "
                                    f"for graph traversal not found for target ObjectType "
                                    f"'{traversal.target_object_type_name}'."
                                )
                            # TODO: Add operator/value type compatibility check

                    component_steps.append(PlannedStep(
                        step_type="kuzu_traversal",
                        details={
                            "source_object_type_name": component_object_type_name,
                            "graph_traversal_clause": traversal
                        },
                        input_object_ids_source_step_index=current_input_source_idx_for_component
                    ))
                    current_input_source_idx_for_component = len(component_steps) - 1
            
            all_component_plans.append(PlannedComponentExecution(
                component_index=idx,
                object_type_name=component_object_type_name,
                steps=component_steps
            ))

        return PlannedQuery(
            original_query=query,
            final_target_object_type_name=final_target_object_type,
            component_plans=all_component_plans
        )

    def execute(self, planned_query: PlannedQuery) -> QueryResult:
        """
        Executes the planned query steps for each component and aggregates results.

        Args:
            planned_query: The PlannedQuery object.

        Returns:
            A QueryResult object.
        """
        logger.info(
            "Executing planned query for: %s",
            planned_query.original_query.description or "Untitled Query"
        )
        
        all_component_final_ids: List[List[UUID]] = [] # Stores final list of Object UUIDs for each component
        errors: List[str] = []

        for comp_plan_idx, component_plan in enumerate(planned_query.component_plans):
            logger.info(
                "Executing component %d (ObjectType: %s)",
                component_plan.component_index, component_plan.object_type_name
            )
            intermediate_step_results: Dict[int, List[UUID]] = {} # IDs per step within this component
            current_component_ids: Optional[List[UUID]] = None

            for step_idx, step in enumerate(component_plan.steps):
                input_ids_for_step: Optional[List[UUID]] = None
                if step.input_object_ids_source_step_index is not None:
                    input_ids_for_step = intermediate_step_results.get(step.input_object_ids_source_step_index)
                    if input_ids_for_step is None:
                        msg = (f"Internal planning error: Missing input IDs for component "
                               f"{comp_plan_idx}, step {step_idx} from source step "
                               f"{step.input_object_ids_source_step_index}.")
                        logger.error(msg)
                        errors.append(msg)
                        # This component fails, skip to next component or return error
                        current_component_ids = [] # Mark as failed for this component
                        break
                    if not input_ids_for_step: # Previous step in this component yielded no results
                        logger.info(
                            "Component %d, Step %d (%s) received no input IDs, short-circuiting component.",
                            comp_plan_idx, step_idx, step.step_type
                        )
                        current_component_ids = []
                        break
                
                try:
                    step_output_ids: List[UUID] = []
                    if step.step_type == "sqlite_filter":
                        details = step.details
                        rel_filters: List[RelationalFilter] = details["filters"]
                        obj_type_name: str = details["object_type_name"]
                        
                        step_output_ids = self._db_manager.sqlite_adapter.find_object_ids_by_properties(
                            object_type_name=obj_type_name,
                            filters=rel_filters,
                            initial_ids=input_ids_for_step # Pass previous step's IDs to refine
                        )
                        logger.debug("SQLite filter step yielded %d IDs", len(step_output_ids))

                    elif step.step_type == "lancedb_search":
                        emb_clause: EmbeddingSearchClause = step.details["embedding_search_clause"]
                        step_output_ids = self._db_manager.lancedb_adapter.find_object_ids_by_similarity(
                            embedding_definition_name=emb_clause.embedding_definition_name,
                            query_vector=emb_clause.similar_to_payload,
                            limit=emb_clause.limit,
                            initial_ids=input_ids_for_step # Pass previous step's IDs to refine
                            # filter_condition might be added later if needed from emb_clause
                        )
                        logger.debug("LanceDB search step yielded %d IDs", len(step_output_ids))

                    elif step.step_type == "kuzu_traversal":
                        trav_clause: GraphTraversalClause = step.details["graph_traversal_clause"]
                        source_obj_type_name: str = step.details["source_object_type_name"]
                        if input_ids_for_step is None: # Kuzu step must have input IDs
                            msg = (f"Kuzu traversal (Comp {comp_plan_idx}, Step {step_idx}) "
                                   f"requires input IDs, but none were provided.")
                            logger.error(msg)
                            errors.append(msg)
                            step_output_ids = []
                        else:
                            step_output_ids = self._db_manager.kuzu_adapter.filter_object_ids_by_relations(
                                source_object_type_name=source_obj_type_name,
                                object_ids=input_ids_for_step,
                                traversals=[trav_clause] # Expects a list
                            )
                        logger.debug("Kuzu traversal step yielded %d IDs", len(step_output_ids))
                    else:
                        msg = f"Unknown query step type '{step.step_type}' at component {comp_plan_idx}, step {step_idx}."
                        logger.error(msg)
                        errors.append(msg)
                        step_output_ids = [] # Mark as failed

                    intermediate_step_results[step_idx] = step_output_ids
                    current_component_ids = step_output_ids # Update with the latest set of IDs for this component

                    if not current_component_ids: # No results after this step
                        logger.info(
                            "Component %d, Step %d (%s) yielded no results. Aborting component execution.",
                            comp_plan_idx, step_idx, step.step_type
                        )
                        break # Break from steps loop for this component

                except Exception as e:
                    msg = (f"Error in component {comp_plan_idx}, step {step_idx} ({step.step_type}): {e}")
                    logger.error(msg, exc_info=True)
                    errors.append(msg)
                    current_component_ids = [] # Mark component as failed
                    break # Break from steps loop for this component
            
            # After all steps for a component are done (or short-circuited)
            if current_component_ids is None: # Should not happen if steps were processed
                current_component_ids = []
            all_component_final_ids.append(current_component_ids)
            logger.info(
                "Component %d finished with %d IDs.",
                component_plan.component_index, len(current_component_ids)
            )

        # Aggregate results from all components (Intersection for AND logic)
        final_aggregated_ids: Optional[List[UUID]] = None
        if not errors: # Only aggregate if no critical errors stopped a component prematurely
            if all_component_final_ids:
                # Start with the IDs from the first component
                final_aggregated_ids_set = set(all_component_final_ids[0])
                # Intersect with subsequent components
                for i in range(1, len(all_component_final_ids)):
                    final_aggregated_ids_set.intersection_update(all_component_final_ids[i])
                final_aggregated_ids = list(final_aggregated_ids_set)
                logger.info("Final aggregated query resulted in %d IDs after intersection.", len(final_aggregated_ids))
            else: # No components or all components failed to produce IDs
                final_aggregated_ids = []
        else: # If there were errors, result is empty or partial (currently empty)
            final_aggregated_ids = []
            logger.warning("Query execution had errors, final ID list is empty.")


        # Fetch full ObjectInstances for the final_aggregated_ids
        final_instances: List[ObjectInstance] = []
        if final_aggregated_ids:
            try:
                # GrizabellaDBManager.get_objects_by_ids should handle UUIDs directly
                final_instances = self._db_manager.get_objects_by_ids(
                    object_type_name=planned_query.final_target_object_type_name,
                    object_ids=final_aggregated_ids
                )
                logger.info("Fetched %d full object instances.", len(final_instances))
            except Exception as e:
                msg = f"Error fetching final object instances: {e}"
                logger.error(msg, exc_info=True)
                errors.append(msg)
                # Potentially return partial results if some IDs were fetched before error
                # For now, if fetching fails, results are empty.

        return QueryResult(object_instances=final_instances, errors=errors if errors else None)
